	#picture = GFX_report_event_ronda_1918
	
	is_triggered_only = yes
	
	trigger = {
		original_tag = ENG
	}
	
	fire_only_once = yes
	
	option = {	#Ceden gibraltar a andalucia
		name = "andalucia.4.o1"
		AND = { country_event = { id = andalucia.5 hours = 15 } }
		#ai_chance = { factor = 100 }
		ai_chance = {
			factor = 2
			modifier = {
				factor = 6
				threat > 0.2
			}
			modifier = {
				factor = 10
				threat > 0.5
			}
			modifier = {
				factor = 15
				threat > 0.7
			}
			modifier = {
				factor = 200
				threat > 0.8
			}
		}
	}
	option = { # Niegan la oferta
		name = "andalucia.4.o2"
		AND = { country_event = { id = andalucia.6 hours = 15 } }
		ai_chance = { factor = 20 }
	}
}
